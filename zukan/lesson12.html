完成版のコード 💻

以下が図鑑Webサービスのすべての機能が入った完成版です。コピーして index.html に使おう！

<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>わたしのオリジナル図鑑</title>
  <style>
    /* 省略 - 前回までのCSSをそのまま使おう */
    body { background-color: #f0f8ff; font-family: sans-serif; margin: 0; }
    header { background-color: #2e8b57; color: white; text-align: center; padding: 20px; }
    .zukan-container { display: flex; flex-wrap: wrap; gap: 20px; justify-content: center; padding: 24px; }
    .card { background: white; width: 250px; padding: 15px; border-radius: 10px;
            box-shadow: 0 4px 8px rgba(0,0,0,0.1); text-align: center; cursor: pointer;
            transition: transform 0.2s; }
    .card:hover { transform: translateY(-4px); }
    .card h2 { color: #ff6347; }
    img { border-radius: 8px; }
    ul { list-style: none; padding: 0; color: #666; font-size: 14px; text-align: left; }
    .add-form { background: white; max-width: 500px; margin: 20px auto; padding: 20px;
                border-radius: 10px; box-shadow: 0 4px 8px rgba(0,0,0,0.1); text-align: center; }
    .add-form input { display: block; width: 80%; margin: 8px auto; padding: 8px 12px;
                      border: 1px solid #ccc; border-radius: 5px; }
    .add-form button { margin-top: 12px; padding: 10px 24px; background: #2e8b57;
                       color: white; border: none; border-radius: 5px; cursor: pointer; }
  </style>
</head>
<body>
  <header>
    <h1>🌿 わたしのオリジナル図鑑 🐛</h1>
  </header>

  <div class="add-form">
    <h2>🐾 生き物を追加する</h2>
    <input id="input-name"  type="text" placeholder="名前">
    <input id="input-place" type="text" placeholder="場所">
    <input id="input-date"  type="text" placeholder="日付">
    <input id="input-memo"  type="text" placeholder="メモ">
    <button onclick="addIkimono()">追加する！</button>
  </div>

  <div id="zukan-container" class="zukan-container"></div>

  <script>
    let zukanData = [
      { name:"アサガオ", description:"夏の花",   place:"学校の花壇", date:"7月20日" },
      { name:"カブトムシ",description:"夏の虫",   place:"近くの森",   date:"8月5日"  },
      { name:"タンポポ",  description:"春の花",   place:"公園",       date:"4月10日" }
    ];

    let container = document.getElementById("zukan-container");

    function renderCard(ikimono) {
      container.innerHTML += `
        <div class="card" onclick="alert('【${ikimono.name}】\n${ikimono.description}\n📍${ikimono.place} 📅${ikimono.date}')">
          <h2>${ikimono.name}</h2>
          <img src="https://placehold.jp/240x180.png?text=${encodeURIComponent(ikimono.name)}"
               alt="${ikimono.name}" width="200">
          <p>${ikimono.description}</p>
          <ul>
            <li>📍 場所：${ikimono.place}</li>
            <li>📅 日付：${ikimono.date}</li>
          </ul>
        </div>
      `;
    }

    zukanData.forEach(renderCard);

    function addIkimono() {
      let name  = document.getElementById("input-name").value;
      let place = document.getElementById("input-place").value;
      let date  = document.getElementById("input-date").value;
      let memo  = document.getElementById("input-memo").value;
      if (name === "") { alert("名前を入力してください！"); return; }
      let newData = { name, description: memo||"（メモなし）", place: place||"（場所なし）", date: date||"（日付なし）" };
      zukanData.push(newData);
      renderCard(newData);
      document.getElementById("input-name").value = "";
      document.getElementById("input-place").value = "";
      document.getElementById("input-date").value = "";
      document.getElementById("input-memo").value = "";
      alert(`「${name}」を追加したよ！🎉`);
    }
  </script>
</body>
</html>
